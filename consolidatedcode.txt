console.log('[Main] Starting standalone WASM test...');

async function init() {
    const statusEl = document.getElementById('status');
    
    try {
        statusEl.textContent = 'Creating shared memory...';
        
        // Create shared memory - 256 pages = 16MB
        const memory = new WebAssembly.Memory({ 
            initial: 256, 
            maximum: 256, 
            shared: true 
        });
        
        console.log('[Main] Shared memory created');
        console.log('[Main] Buffer is SharedArrayBuffer:', memory.buffer instanceof SharedArrayBuffer);
        console.log('[Main] Buffer size:', memory.buffer.byteLength);
        
        // Initialize memory to zero
        const view = new DataView(memory.buffer);
        const RADIX_HEAD_OFFSET = 0x100;
        view.setBigUint64(RADIX_HEAD_OFFSET, 0n, true);
        console.log('[Main] Memory initialized to 0 at offset 0x100');
        
        statusEl.textContent = 'Loading WASM bytes for worker...';
        
        // Get the WASM bytes to send to worker
        const wasmPath = './pkg/radixrunner_bg.wasm';
        const response = await fetch(wasmPath);
        const wasmBytes = await response.arrayBuffer();
        
        console.log('[Main] WASM bytes loaded:', wasmBytes.byteLength);
        
        statusEl.textContent = 'Spawning tick worker...';
        
        // Spawn tick worker
        const tickWorker = new Worker('./tick_worker.js', { type: 'module' });
        
        tickWorker.postMessage({
            cmd: 'init',
            memory: memory,
            wasmBytes: wasmBytes
        });
        
        tickWorker.onerror = (err) => {
            console.error('[Main] Tick worker error:', err);
            statusEl.textContent = 'ERROR: Tick worker failed - ' + err.message;
            statusEl.classList.add('error');
        };
        
        tickWorker.onmessage = (e) => {
            if (e.data.type === 'error') {
                console.error('[Main] Tick worker error:', e.data.error);
                console.error('[Main] Stack:', e.data.stack);
                statusEl.textContent = 'ERROR: ' + e.data.error;
                statusEl.classList.add('error');
            } else if (e.data.type === 'ready') {
                console.log('[Main] Tick worker ready');
                
                // Check memory immediately
                const view = new DataView(memory.buffer);
                const immediate = view.getBigUint64(RADIX_HEAD_OFFSET, true);
                console.log('[Main] Memory check immediately after ready:', '0x' + immediate.toString(16));
                
                // Wait for some ticks to accumulate
                setTimeout(() => {
                    const testValue = view.getBigUint64(RADIX_HEAD_OFFSET, true);
                    console.log('[Main] Memory check after 100ms:', '0x' + testValue.toString(16));
                    
                    if (testValue > 0n) {
                        console.log('[Main] ✅ Memory is being updated!');
                        console.log('[Main] Tick rate: ~', Number(testValue) / 0.1, 'ticks/second');
                        statusEl.textContent = '✅ WASM Runtime Active - Clock Running';
                        statusEl.classList.remove('error');
                        statusEl.classList.add('success');
                        startMonitoring(memory);
                    } else {
                        console.error('[Main] ❌ Memory still at 0 - shared memory not working');
                        statusEl.textContent = '❌ ERROR: Memory not shared between threads';
                        statusEl.classList.add('error');
                        
                        // Additional debugging
                        console.log('[Main] Memory buffer address:', memory.buffer);
                        console.log('[Main] Is SharedArrayBuffer?', memory.buffer instanceof SharedArrayBuffer);
                        console.log('[Main] Buffer byteLength:', memory.buffer.byteLength);
                    }
                }, 100);
            }
        };
        
    } catch (err) {
        console.error('[Main] Initialization failed:', err);
        console.error('[Main] Stack:', err.stack);
        statusEl.textContent = '❌ ERROR: ' + err.message;
        statusEl.classList.add('error');
    }
}

function startMonitoring(memory) {
    const RADIX_HEAD_OFFSET = 0x100;
    
    console.log('[Main] Starting monitor');
    console.log('[Main] Memory buffer size:', memory.buffer.byteLength);
    console.log('[Main] Reading from offset:', '0x' + RADIX_HEAD_OFFSET.toString(16));
    
    let lastValue = 0n;
    let lastTime = performance.now();
    
    function update() {
        try {
            const view = new DataView(memory.buffer);
            const full = view.getBigUint64(RADIX_HEAD_OFFSET, true);
            
            // Calculate tick rate
            const now = performance.now();
            const deltaTime = now - lastTime;
            if (deltaTime > 1000) { // Update rate calculation every second
                const deltaValue = full - lastValue;
                const ticksPerSecond = Number(deltaValue) / (deltaTime / 1000);
                console.log('[Main] Tick rate:', ticksPerSecond.toFixed(0), 'ticks/second');
                lastValue = full;
                lastTime = now;
            }
            
            // Extract precision components
            const p0 = Number(full & 0xFFFn);
            const p1 = Number((full >> 12n) & 0x3FFn);
            const p2 = Number((full >> 22n) & 0x3FFn);
            const p3 = Number((full >> 32n) & 0x3FFn);
            const p4 = Number((full >> 42n) & 0x3FFn);
            const p5 = Number((full >> 52n) & 0xFFFn);
            
            // Update display
            document.getElementById('p0').textContent = p0;
            document.getElementById('p1').textContent = p1;
            document.getElementById('p2').textContent = p2;
            document.getElementById('p3').textContent = p3;
            document.getElementById('p4').textContent = p4;
            document.getElementById('p5').textContent = p5;
            document.getElementById('full').textContent = '0x' + full.toString(16).padStart(16, '0');
            
            requestAnimationFrame(update);
        } catch (err) {
            console.error('[Main] Monitor error:', err);
            requestAnimationFrame(update);
        }
    }
    
    update();
}

// src/memory_worm.rs
use std::sync::atomic::{AtomicU64, Ordering};

// Memory layout constants
pub const WORM_HEAD_OFFSET: usize = 0x0000;
pub const RADIX_HEAD_OFFSET: usize = 0x0100;
pub const APP_SECTIONS_OFFSET: usize = 0x1000;

// Bit layout [P5:12][P4:10][P3:10][P2:10][P1:10][P0:12]
pub const P0_BITS: u32 = 12; pub const P0_MASK: u64 = 0xFFF; pub const P0_SHIFT: u32 = 0;
pub const P1_BITS: u32 = 10; pub const P1_MASK: u64 = 0x3FF; pub const P1_SHIFT: u32 = 12;
pub const P2_BITS: u32 = 10; pub const P2_MASK: u64 = 0x3FF; pub const P2_SHIFT: u32 = 22;
pub const P3_BITS: u32 = 10; pub const P3_MASK: u64 = 0x3FF; pub const P3_SHIFT: u32 = 32;
pub const P4_BITS: u32 = 10; pub const P4_MASK: u64 = 0x3FF; pub const P4_SHIFT: u32 = 42;
pub const P5_BITS: u32 = 12; pub const P5_MASK: u64 = 0xFFF; pub const P5_SHIFT: u32 = 52;

// Global base pointer - raw and direct
static mut WORM_BASE: *mut u8 = std::ptr::null_mut();

pub struct MemoryWorm;

impl MemoryWorm {
    #[inline]
    pub unsafe fn init(base: *mut u8) {
        WORM_BASE = base;
        
        // Verify we can access the memory (no logging)
        let ptr = WORM_BASE.add(RADIX_HEAD_OFFSET) as *mut AtomicU64;
        let _initial = (*ptr).load(Ordering::Relaxed);
        
        // JavaScript will verify the memory is accessible
    }
    
    #[inline]
    pub fn radix_head() -> &'static AtomicU64 {
        unsafe {
            let ptr = WORM_BASE.add(RADIX_HEAD_OFFSET) as *mut AtomicU64;
            &*ptr
        }
    }
    
    #[inline]
    pub fn tick() {
        Self::radix_head().fetch_add(1, Ordering::Relaxed);
    }
    
    #[inline]
    pub fn full() -> u64 {
        Self::radix_head().load(Ordering::Relaxed)
    }
    
    #[inline]
    pub fn section_start(id: u32) -> *mut u8 {
        unsafe {
            WORM_BASE.add(APP_SECTIONS_OFFSET + (id as usize * 0x1000))
        }
    }
}
